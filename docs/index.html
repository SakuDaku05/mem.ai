Quick Start

Integrate in seconds

Start the server and immediately connect your agents.

$ pip install "memai[all]"
Python SDK
CLI Usage
from memai.sdk import MemaiClient

# Connect to your local or remote memai server
mem = MemaiClient(api_key="sk-memai-...", base_url="http://localhost:8000")

# Store facts
mem.add("User prefers concise answers", agent_id="agent-1")

# Search and inject optimally
context = mem.inject("What are the user's preferences?", agent_id="agent-1")

# Causal session tracking
with mem.session("agent-1") as s:
    e1 = s.add_event("Data loaded")
    e2 = s.add_event("Error occurred", caused_by=e1.id)
